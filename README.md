```python
def search_word(grid, word):
    """
    ğŸ¯ Searches for a word in all 8 directions
    Returns: List of matches with full path coordinates
    """
    rows, cols = len(grid), len(grid[0])
    word = word.upper()
    
    # 8-direction vectors
    directions = {
        "RIGHT": (0, 1),      "LEFT": (0, -1),
        "DOWN": (1, 0),       "UP": (-1, 0),
        "DOWN-RIGHT": (1, 1), "UP-LEFT": (-1, -1),
        "DOWN-LEFT": (1, -1), "UP-RIGHT": (-1, 1)
    }
    
    def check_direction(start_x, start_y, dx, dy):
        path = []
        for i in range(len(word)):
            x, y = start_x + i * dx, start_y + i * dy
            if not (0 <= x < rows and 0 <= y < cols):
                return None
            if grid[x][y].upper() != word[i]:
                return None
            path.append((x, y))
        return path
    
    results = []
    for i in range(rows):
        for j in range(cols):
            if grid[i][j].upper() == word[0]:  # First letter match
                for dir_name, (dx, dy) in directions.items():
                    path = check_direction(i, j, dx, dy)
                    if path:
                        results.append({
                            "word": word,
                            "start": (i, j),
                            "direction": dir_name,
                            "path": path,
                            "length": len(word)
                        })
    
    return results if results else [{"word": word, "found": False}]
```
